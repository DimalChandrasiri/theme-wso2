var data = [ { "metadata": { "names": ["rpm", "torque", "horsepower", "EngineType"], "types": ["ordinal", "linear", "ordinal", "ordinal"] }, "data": [] } ]; var config = { type: "bar", x: "rpm", highlight: "multi", selectionColor: "#33ccff", charts: [{type: "bar", y: "torque"}], maxLength: 6, width: 500, height: 250 } var barChart = new vizg(data, config); barChart.draw("#bChart"); var iterations = 1000; (function insertLoop(i) { setTimeout(function () { var xVal = Number((Math.random() * 100).toFixed(0)) % 15; barChart.insert([[xVal, Number((Math.random() * 100).toFixed(0)) % 15, 20, "Piston"]]); if (--i) insertLoop(i); }, 1000) })(iterations);

Donut Chart

var data = [ { "metadata" : { "names" : ["rpm","torque","horsepower", "EngineType"], "types" : ["linear","linear", "ordinal","ordinal"] }, "data": [ [0, 100, 12,"Rotary"] ] } ]; var configDonut = { charts : [{type: "arc", x : "torque", color : "EngineType", mode: "donut"}], width: 400; height: 300, percentage:true } var donutChart = new vizg(data, configDonut); donutChart.draw("#dChart"); var iterations = 1000; (function insertLoop (i) { setTimeout(function () { if ( i % 4 == 0) { EngineType = "Rotary"; } else if( i % 4 == 1) { EngineType = "Piston V6"; } else if( i % 4 == 2) { EngineType = "Piston V8"; } else if( i % 4 == 3) { EngineType = "Piston"; } donutChart.insert([[iterations-i + 1, Number((Math.random() * 100).toFixed(2)), 13, EngineType]]); if (--i) insertLoop(i); }, 1000) })(iterations);
